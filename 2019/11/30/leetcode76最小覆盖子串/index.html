    题目描述

    给你一个字符串 S、一个字符串 T，请在字符串 S 里面找出：包含 T 所有字母的最小子串。

    示例：

    输入: S = "ADOBECODEBANC", T = "ABC"
输出: "BANC"

    说明：

    如果 S 中不存这样的子串，则返回空字符串 “”。
    如果 S 中存在这样的子串，我们保证它是唯一的答案。

    来源：力扣（LeetCode）
    链接：https://leetcode-cn.com/problems/minimum-window-substring

    题解

    注意这道题干中包含T所有字母的最小子串，这里的字母是可重复的，一开始博主做题时就会错题意而解答错误了。

    参考链接：
    https://blog.csdn.net/qq_37206105/article/details/88653767

    class Solution {
public:
    string minWindow(string s, string t) {
        if(s.size() < t.size()) return "";

        int sFreq[64] = {0}, tFreq[64] = {0};
        for(int i = 0; i < t.size(); i++)
            tFreq[t[i] - 'A']++;

        int l = 0, r = -1, edge[2] = {-1, s.size() + 1};        // edge数组表示要求的串的左右边界
        while(l <= s.size() - t.size())                     // 若大于这个边界，则不满足
        {
            /*一开始右边界右移，size 没达到时，直接右移，循环continue*/
            if(r - l + 1 < t.size())
            {
                //if(r + 1 < s.size())              //这里是考虑数组越界，但是可以不写 ，因为两个限定条件已经保证他不会越界
                //{
                    sFreq[s[++r] - 'A']++;
                    continue;
                //}
                //else break;
            }

            /* 判断现在窗口中的字符是否满足“题目要求” */
            int i = 0;
            while(i < 64)
            {
                if(sFreq[i] < tFreq[i]) break;
                i++;
            }
            if(i < 64)          // 不满足要求
            {
                if(r + 1 < s.size()) sFreq[s[++r] - 'A']++;
                else break;
            }
            else
            {
                if (r - l + 1 == t.size()) 
                {
                    string str = s.substr(l, r - l + 1);
                    return str;
                }
                else
                {
                    if(r - l < edge[1] - edge[0])
                    {
                        edge[0] = l;
                        edge[1] = r;
                    }
                    sFreq[s[l++] - 'A']--;
                }
            }
        }
        if (edge[0] == -1)
        {
            return "";
        }
        else
        {
            string str = s.substr(edge[0], edge[1] - edge[0] + 1);
                    return str;
        }
    }
};

    string minWindow(string s, string t) {
    int count[256] = { 0 };
    for (auto c : t) ++count[c];
    int len = 0, minLength = s.length();
    string res;
    for (int l = 0, r = 0; r < s.length(); ++r) {
        count[s[r]]--;
        if (count[s[r]] >= 0) ++len;                        // 这里很精彩，以此区分是否是t中字符
        while (len == t.length()) {
            if (r - l + 1 <= minLength) {
                minLength = r - l + 1;
                res = s.substr(l, r - l + 1);
            }
            count[s[l]]++;
            if (count[s[l]] > 0) --len;                         //若l代表的字符是t中字符，要--len
            ++l;
        }
    }
    return res;
}

    再压缩一下：

    string minWindow(string s, string t) {
    int count[256] = { 0 };
    for (auto c : t) ++count[c];
    int len = 0, minLength = s.length();
    string res;
    for (int l = 0, r = 0; r < s.length(); ++r) {
        if (--count[s[r]] >= 0) ++len;
        while (len == t.length()) {
            if (r - l + 1 <= minLength) {
                minLength = r - l + 1;
                res = s.substr(l, r - l + 1);
            }
            if (++count[s[l++]] > 0) --len;
        }
    }
    return res;
}
