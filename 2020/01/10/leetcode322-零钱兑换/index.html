    题目描述

    给定不同面额的硬币 coins 和一个总金额 amount。编写一个函数来计算可以凑成总金额所需的最少的硬币个数。如果没有任何一种硬币组合能组成总金额，返回 -1。

    示例 1:

    输入: coins = [1, 2, 5], amount = 11
输出: 3 
解释: 11 = 5 + 5 + 1

    示例 2:

    输入: coins = [2], amount = 3
输出: -1

    题解

    这是完全背包问题，首先想到使用动态规划

    class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        if(!coins.size()) return -1;
        vector<int> dp(amount + 1, INT_MAX);
        dp[0] = 0;
        for(int i = 0; i < coins.size(); i++)                   // 其实对 amount来考虑会更好
        {
            for(int j = coins[i]; j <= amount; j++)
            {
                if(dp[j - coins[i]] != INT_MAX)                 // 有点哈希表的味道
                {
                    dp[j] = min(dp[j], dp[j - coins[i]] + 1);  // 用coin(i)时，各价值所用的硬币最少。
                }
            }
        }
        if(dp[amount] != INT_MAX)
        {
            return dp[amount];          // dp[11] = dp[1] + coin[2] + coin[2]
        }
        return -1;
    }
};

    实际上使用总金额数作为外层循环好像更好理解

    class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        vector<int> dp(amount+1,amount+1);
        dp[0]=0;
        for(int i=1;i<=amount;i++){
            for(int c:coins){
                if(i>=c) dp[i] = min(dp[i],dp[i-c]+1);
            }
        }
        return dp[amount]==amount+1?-1:dp[amount];
    }
};

    最后是深度优先搜索，注意，直觉告诉我们 贪心是不可行的

    class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        sort(coins.begin(), coins.end());
        dfs(coins, amount, 0, coins.size() - 1);
        return minCount == INT_MAX? -1 : minCount;
    }
    int minCount = INT_MAX;
    void dfs(vector<int> & coins, int amount, int count, int index)
    {
        //if(index < 0 || minCount != INT_MAX) return;  
        if(index < 0 || count + amount / coins[index] >= minCount) return;     // 直觉也告诉我们，贪心是不可行的，见少一条代码已测试
        if(amount % coins[index] == 0)
        {
            minCount = min(minCount, count + amount / coins[index]);
            return;
        }          
        for(int i = amount / coins[index]; i >= 0; i--)
        {
            dfs(coins, amount - i * coins[index], count + i, index - 1);
        }             
    }
};
